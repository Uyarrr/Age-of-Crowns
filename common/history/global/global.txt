GLOBAL = {
    every_state = {
        limit = {
            owner = {
                OR = {
                    country_tier = principality
                    country_tier = grand_principality
                    country_tier = kingdom
                    country_tier = empire
                }
            }
        }

        # Fixes Missing Port Connections
        if = {
            limit = {
                is_sea_adjacent = yes
                OR = {
                    is_isolated_from_market = yes
                    is_capital = yes
                }
            }
            create_building = { building = building_port level = 1 }
        }

        # Basic Buildings
        if = {
            limit = { is_capital = yes }
            create_building = { building = building_village  level = 1 }
        }
    }
 	
    every_country = {
 		add_modifier = {
			name = organic_development
			years = 5
		}
    }
}
