# Minimal replacement for ai_strategy_default
# Cleaned for total-conversion / alt-timeline mods.
# Keeps generic AI backbone only.

ai_strategy_default = {
	icon = "gfx/interface/icons/ai_strategy_icons/placate_population.dds" # Never shown

	# Leave empty unless your mod uses power blocs
	will_form_power_bloc = {}

	# Tax behavior
	desired_tax_level = medium
	max_tax_level = high
	min_tax_level = low

	# Expansion / infamy safety rails
	undesirable_infamy_level = {
		value = 50
	}
	unacceptable_infamy_level = {
		value = 100
	}

	# Ideological diplomacy
	ideological_opinion_effect_mult = {
		value = 1.0
	}

	# Law / revolution behavior
	revolution_aversion = {
		value = 5
	}

	min_law_chance_to_pass = {
		value = 50
	}

	negotiation_chance = {
		value = 0

		if = {
			limit = {
				scope:interest_group ?= {
					amenability_level != 4
				}
			}
			value = scope:amenability

			add = scope:stall_chance

			if = {
				limit = {
					ai_has_reasons_to_not_oppose_law = yes
				}
				multiply = 2
			}

			subtract = {
				value = scope:advance_chance
				multiply = 2
			}
		}

		max = 1
	}

	max_progressiveness = {
		value = 25
	}

	max_regressiveness = {
		value = 25
	}

	# Generic diplomatic play support only
	diplomatic_play_support = {
		value = 0

		add = {
			value = "scope:country.modifier:country_diplomatic_reputation_add"
			desc = "DIPLOMATIC_REPUTATION_REASON"
		}
	}

	diplomatic_play_neutrality = {
		value = 50
	}

	diplomatic_play_boldness = {
		value = 50

		# Generic fervor / zeal style behavior if your mod uses this value
		if = {
			limit = {
				country_fervor_primary_culture > 10
			}
			add = {
				value = country_fervor_primary_culture
				divide = 4
			}
		}
	}

	wargoal_maneuvers_fraction = {
		value = 0.35

		if = {
			limit = {
				scope:enemy_country.country_rank > country_rank
			}
			add = -0.1
		}
	}

	change_law_chance = {
		value = 1
	}

	pro_interest_groups = {}
	anti_interest_groups = {}

	# Keep empty if your institution setup is heavily changed.
	# Add your custom institutions later.
	institution_scores = {}

	# Generic obligation valuation
	obligation_value = {
		value = 5

		if = {
			limit = {
				scope:target_country.country_rank > country_rank
			}
			add = {
				value = scope:target_country.country_rank
				multiply = 5
			}
			subtract = {
				value = country_rank
				multiply = 5
			}
		}
		else_if = {
			limit = {
				scope:target_country.country_rank < country_rank
			}
			add = {
				value = scope:target_country.country_rank
				multiply = 2
			}
			subtract = {
				value = country_rank
				multiply = 2
			}
		}

		if = {
			limit = { scope:target_country.country_rank = rank_value:great_power }
			add = 5
		}

		if = {
			limit = {
				OR = {
					has_attitude = {
						who = scope:target_country
						attitude = protective
					}
					has_attitude = {
						who = scope:target_country
						attitude = domineering
					}
				}
			}
			add = 15
		}

		if = {
			limit = {
				scope:target_country = {
					OR = {
						has_attitude = {
							who = root
							attitude = belligerent
						}
						has_attitude = {
							who = root
							attitude = domineering
						}
					}
				}
			}
			add = 10
		}

		if = {
			limit = {
				OR = {
					has_attitude = {
						who = scope:target_country
						attitude = genial
					}
					has_attitude = {
						who = scope:target_country
						attitude = cooperative
					}
				}
			}
			add = 5
		}

		multiply = {
			value = "scope:target_country.relations:root"
			multiply = 0.01
			add = 1
			min = 0.5
			max = 2.0
		}

		multiply = {
			value = "scope:target_country.ai_army_comparison(root)"
			divide = 10
			add = 1
			min = 0.5
			max = 2.0
		}

		multiply = {
			value = "ai_ideological_opinion(scope:target_country)"
			multiply = 0.025
			add = 1
			min = 0.5
			max = 1.5
		}

		if = {
			limit = {
				has_attitude = {
					who = scope:target_country
					attitude = disinterested
				}
			}
			multiply = 0.0
		}
		else_if = {
			limit = {
				has_attitude = {
					who = scope:target_country
					attitude = cautious
				}
			}
			multiply = 0.25
		}

		if = {
			limit = { scope:target_country = { is_subject_of = root } }
			add = 10
		}

		min = 0
		max = 30
	}

	interest_group_government_weight = {
		value = 0
	}

	# Generic state valuation
	state_value = {
		value = 0

		if = {
			limit = {
				OR = {
					scope:target_state = { owner = root }
					can_reach_target_state = yes
				}
			}

			if = {
				limit = {
					OR = {
						scope:target_state = { owner = root }
						ai_desires_target_state = yes
					}
				}

				add = {
					value = 20

					add = {
						value = scope:target_state.state_population
						divide = 100000
						max = 20
					}

					add = {
						value = scope:target_state.gdp
						divide = 100000
						max = 20
					}
				}

				if = {
					limit = { scope:target_state = { is_treaty_port = yes } }
					add = 20
				}

				if = {
					limit = {
						country_fervor_primary_culture > 10
						scope:target_state = {
							is_homeland_of_country_cultures = root
						}
						scope:target_country = {
							NOT = {
								any_primary_culture = {
									is_primary_culture_of = root
								}
							}
						}
					}
					add = {
						value = country_fervor_primary_culture
						divide = 2
					}
				}
			}

			if = {
				limit = {
					scope:target_state = { owner = root }
				}

				if = {
					limit = {
						scope:target_state = { ai_can_incorporate_state = yes }
					}
					add = 50
				}

				if = {
					limit = {
						scope:target_state = {
							OR = {
								is_incorporated = yes
								ai_can_incorporate_state = yes
							}
						}
					}

					if = {
						limit = {
							scope:target_state = { is_homeland_of_country_cultures = root }
						}
						multiply = 8.0
					}
					else = {
						multiply = 6.0
					}
				}
				else_if = {
					limit = {
						scope:target_state = {
							OR = {
								any_neighbouring_state = {
									owner = root
									is_incorporated = yes
								}
								any_sea_node_adjacent_state = {
									owner = root
									is_incorporated = yes
								}
							}
						}
					}

					if = {
						limit = {
							scope:target_state = { is_homeland_of_country_cultures = root }
						}
						multiply = 6.0
					}
					else = {
						multiply = 4.0
					}
				}
				else = {
					if = {
						limit = {
							scope:target_state = { is_homeland_of_country_cultures = root }
						}
						multiply = 3.0
					}
					else = {
						multiply = 2.0
					}
				}

				if = {
					limit = {
						is_at_war = no
						scope:target_state = { is_isolated_from_market = yes }
					}
					multiply = 0.5

					if = {
						limit = {
							NOT = {
								any_neighbouring_state = { owner = root }
							}
						}
						multiply = 0.5
					}
				}
			}
			else = {
				if = {
					limit = {
						OR = {
							scope:target_state = {
								is_split_state = yes
								state_region = {
									any_scope_state = {
										owner = ROOT
										is_treaty_port = no
									}
								}
							}
							scope:target_state = { has_claim_by = root }
						}
					}
					multiply = 2.0
				}

				if = {
					limit = {
						NOT = { has_strategic_land_adjacency = scope:target_state }
						scope:target_state = { has_port = no }
					}
					multiply = 0.5
				}

				if = {
					limit = {
						has_strategic_land_adjacency = scope:target_country
						NOT = { has_strategic_land_adjacency = scope:target_state }
					}
					multiply = 0.5
				}
				else_if = {
					limit = {
						is_adjacent_to_country = scope:target_country
						NOT = { is_adjacent_to_state = scope:target_state }
					}
					multiply = 0.75
				}

				if = {
					limit = {
						is_diplomatic_play_involved_with = scope:target_country
						ai_strongly_desires_target_state = no
					}
					multiply = 0.5
				}

				if = {
					limit = {
						scope:target_state = { is_homeland_of_country_cultures = root }
					}
					multiply = 1.5
				}
			}
		}

		min = 0
		if = {
			limit = { scope:target_state = { owner = root } }
			max = 1000
		}
		else = {
			max = 200
		}
	}

	# Generic treaty port valuation
	treaty_port_value = {
		value = 0

		if = {
			limit = {
				scope:target_state = {
					state_region = {
						any_scope_state = {
							is_treaty_port = yes
							OR = {
								owner = { is_subject_of = ROOT }
								owner = ROOT
							}
						}
					}
				}
			}
			add = 40
		}

		if = {
			limit = {
				NOT = { is_country_type = unrecognized }
				scope:target_country = { is_country_type = unrecognized }
				NOT = { is_adjacent_to_country = scope:target_country }
				OR = {
					scope:target_country = {
						has_diplomatic_pact = {
							who = root
							type = embargo
						}
					}
					AND = {
						has_strategy = ai_strategy_economic_imperialism
						"mutual_trade_value_with_country(root)" >= 1000
						OR = {
							has_attitude = {
								who = scope:target_country
								attitude = belligerent
							}
							has_attitude = {
								who = scope:target_country
								attitude = domineering
							}
							has_attitude = {
								who = scope:target_country
								attitude = antagonistic
							}
						}
					}
				}
			}

			add = {
				value = 5

				add = {
					value = scope:target_state.state_population
					divide = 200000
					max = 10
				}

				add = {
					value = scope:target_state.gdp
					divide = 200000
					max = 10
				}

				if = {
					limit = {
						any_scope_building = {
							is_building_type = building_trade_center
							level > 0
						}
					}
					add = 5
				}

				if = {
					limit = {
						has_strategic_adjacency = scope:target_state
					}
					multiply = 1.50
				}

				if = {
					limit = {
						scope:target_state = { is_incorporated = no }
					}
					multiply = 0.25
				}
			}
		}

		if = {
			limit = {
				scope:target_state = {
					NOT = {
						state_region = {
							any_scope_state = {
								is_treaty_port = yes
								OR = {
									owner = { is_subject_of = ROOT }
									owner = ROOT
								}
							}
						}
					}
				}
			}

			if = {
				limit = { has_treaty_port_in_market = scope:target_country.market }
				multiply = 0.25
			}

			if = {
				limit = {
					market_capital.market ?= scope:target_country.market_capital.market
				}
				multiply = 0.0
			}

			if = {
				limit = {
					scope:target_country = {
						gdp < 2000000
					}
				}
				multiply = 0
			}

			if = {
				limit = {
					scope:target_country = {
						any_scope_treaty = {
							binds = root
							any_scope_article = {
								has_type = trade_privilege
								source_country = scope:target_country
							}
						}
					}
				}
				multiply = 0.25
			}

			if = {
				limit = {
					can_reach_target_state = no
				}
				multiply = 0
			}
		}

		min = 0
		max = 100
	}

	# Generic subject valuation
	subject_value = {
		value = 0

		if = {
			limit = {
				OR = {
					scope:target_country = { is_subject_of = root }
					can_reach_target_country = yes
				}
			}

			add = {
				if = {
					limit = {
						OR = {
							scope:target_country = { is_subject_of = root }
							has_attitude = {
								who = scope:target_country
								attitude = protective
							}
							has_attitude = {
								who = scope:target_country
								attitude = domineering
							}
						}
					}

					value = 50

					add = {
						value = scope:target_country.gdp
						divide = 200000
						max = 50
					}
				}

				if = {
					limit = {
						scope:target_country = { is_subject_of = root }
					}

					if = {
						limit = {
							scope:target_country = {
								prestige >= 100
							}
						}
						multiply = 5.0
					}
					else_if = {
						limit = {
							scope:target_country = {
								prestige >= 75
							}
						}
						multiply = 4.0
					}
					else_if = {
						limit = {
							scope:target_country = {
								prestige >= 50
							}
						}
						multiply = 3.0
					}
					else_if = {
						limit = {
							scope:target_country = {
								prestige >= 25
							}
						}
						multiply = 2.0
					}
					else = {
						multiply = 1.5
					}
				}
				else = {
					if = {
						limit = {
							has_secret_goal = {
								who = scope:target_country
								secret_goal = protect
							}
						}
						multiply = 0.5
					}

					if = {
						limit = {
							NOT = { has_strategic_adjacency = scope:target_country }
						}
						multiply = 0.5
					}
				}
			}

			if = {
				limit = {
					lenient_ai_behavior_trigger = yes
				}
				multiply = 0.5
			}

			if = {
				limit = {
					AND = {
						has_radical_leftist_government = no
						scope:target_country = {
							has_radical_leftist_government = yes
						}
					}
				}
				multiply = 1.5
			}

			if = {
				limit = {
					AND = {
						has_radical_leftist_economy = no
						scope:target_country = {
							has_radical_leftist_economy = yes
						}
					}
				}
				multiply = 1.5
			}
		}

		if = {
			limit = {
				has_strategy = ai_strategy_economic_imperialism
				scope:target_country = {
					has_law_or_variant = law_type:law_isolationism
					gdp >= 5000000
				}
			}
			multiply = 0.0
		}

		if = {
			limit = {
				scope:target_country = { is_subject_of = root }
			}
			min = 100
			max = 1000
		}
		else_if = {
			limit = {
				has_secret_goal = {
					who = scope:target_country
					secret_goal = dominate
				}
			}
			min = 50
			max = 200
		}
		else = {
			min = 0
			max = 200
		}
	}
}
