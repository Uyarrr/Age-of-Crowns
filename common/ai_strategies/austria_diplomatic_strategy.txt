ai_strategy_hre_imperial_hegemony = {
	icon = "gfx/interface/icons/ai_strategy_icons/maintain_power_balance.dds"

	type = diplomatic

	# Austria begins as leader of the historical HRE bloc. This strategy maintains
	# and expands that bloc rather than creating a replacement.
	will_form_power_bloc = {
		always = no
	}

	undesirable_infamy_level = { value = 25 }
	unacceptable_infamy_level = { value = 50 }
	ideological_opinion_effect_mult = { value = 0.75 }
	wargoal_maneuvers_fraction = { value = 0.10 }
	diplomatic_play_neutrality = { value = -20 }
	diplomatic_play_boldness = { value = 40 }
	recklessness = { value = 0.50 }

	# Defend cordial imperial members against countries outside the Empire.
	# Internal wars and members with poor relations do not receive automatic aid.
	diplomatic_play_support = {
		value = 0

		if = {
			limit = {
				NOT = { exists = scope:is_initiator }
				scope:country = {
					is_in_same_power_bloc = root
					relations:root >= relations_threshold:cordial
				}
				scope:enemy_country = {
					NOT = { is_in_same_power_bloc = root }
				}
			}
			add = 350
		}

		# Never help an outsider launch a play against a friendly HRE member.
		if = {
			limit = {
				exists = scope:is_initiator
				scope:country = {
					NOT = { is_in_same_power_bloc = root }
				}
				scope:enemy_country = {
					is_in_same_power_bloc = root
					relations:root >= relations_threshold:cordial
				}
			}
			add = -1000
		}

		# Do not take sides with an outsider in an internal imperial conflict.
		if = {
			limit = {
				scope:country = { NOT = { is_in_same_power_bloc = root } }
				scope:enemy_country = { is_in_same_power_bloc = root }
			}
			add = -250
		}
	}

	# Austria is diplomatic inside the HRE and only mildly aggressive outside it.
	aggression = {
		value = 0.25

		if = {
			limit = { is_in_same_power_bloc = scope:target_country }
			multiply = 0
		}

		if = {
			limit = { scope:target_country = { is_subject_of = root } }
			multiply = 0.05
		}

		if = {
			limit = {
				has_strategic_adjacency = scope:target_country
				scope:target_country.country_rank <= rank_value:minor_power
				scope:target_country = {
					is_subject = no
					is_in_power_bloc = no
				}
			}
			add = 0.35
		}

		multiply = {
			value = "ai_ideological_opinion(scope:target_country)"
			multiply = -0.05
			add = 1
			min = 0.75
			max = 1.25
		}

		max = 1.0
	}

	building_group_weights = {
		bg_army = 1.40
		bg_technology = 1.20
		bg_navy = 0.50
	}

	secret_goal_weights = {
		protect = 3.0
		befriend = 3.0
		reconcile = 2.0
		dominate = 1.25
		conquer = 0.10
	}

	# Encourage defensive agreements and every diplomatic route that can add
	# countries to the imperial power bloc.
	treaty_category_scores = {
		economy = { value = 4 }
		trade = { value = 5 }
		military = { value = 3 }
		military_defense = { value = 8 }
		ideology = { value = 1 }
		expansion = { value = 3 }
		power_bloc = { value = 10 }
		other = { value = 1 }
		none = { value = 0 }
	}

	wargoal_weights = {
		join_power_bloc = 5.0
		make_protectorate = 1.0
		make_dominion = 0.25
		reduce_autonomy = 1.5
		annex_country = 0.50
		liberate_subject = 1.5
		liberate_country = 1.25
		return_state = 0.75
		conquer_state = 0.10
		force_nationalization = 0.0
	}

	secret_goal_scores = {
		protect = {
			trigger = {
				is_in_same_power_bloc = scope:target_country
				scope:target_country = {
					relations:root >= relations_threshold:cordial
				}
			}

			score = {
				value = 500

				if = {
					limit = {
						scope:target_country = {
							relations:root >= relations_threshold:amicable
						}
					}
					add = 150
				}

				if = {
					limit = { scope:target_country = { is_subject_of = root } }
					add = 100
				}
			}
		}

		# Keep improving relations with every member until the imperial interior
		# is friendly and cohesive.
		befriend = {
			trigger = {
				OR = {
					is_in_same_power_bloc = scope:target_country
					AND = {
						has_strategic_adjacency = scope:target_country
						scope:target_country.country_rank <= rank_value:minor_power
						scope:target_country = {
							is_subject = no
							is_in_power_bloc = no
							any_scope_state = {
								OR = {
									is_in_geographic_region = geographic_region_western_europe
									is_in_geographic_region = geographic_region_central_europe
									is_in_geographic_region = geographic_region_southern_europe
								}
							}
						}
					}
				}
			}

			score = {
				value = 100

				if = {
					limit = { is_in_same_power_bloc = scope:target_country }
					add = 350

					if = {
						limit = {
							scope:target_country = {
								relations:root < relations_threshold:cordial
							}
						}
						add = 300
					}

					if = {
						limit = {
							scope:target_country = {
								relations:root < relations_threshold:friendly
							}
						}
						add = 100
					}
				}

				if = {
					limit = { NOT = { is_in_same_power_bloc = scope:target_country } }
					add = 250

					if = {
						limit = {
							scope:target_country = {
								relations:root < relations_threshold:amicable
							}
						}
						add = 200
					}
				}
			}
		}

		reconcile = {
			trigger = {
				is_in_same_power_bloc = scope:target_country
				scope:target_country = {
					relations:root < relations_threshold:cordial
				}
			}
			score = { value = 400 }
		}

		# Coercion is a fallback for nearby, unaligned imperial candidates that
		# cannot be brought into the bloc through friendly diplomacy.
		dominate = {
			trigger = {
				has_strategic_adjacency = scope:target_country
				scope:target_country.country_rank <= rank_value:minor_power
				scope:target_country = {
					is_subject = no
					is_in_power_bloc = no
					relations:root < relations_threshold:cordial
					any_scope_state = {
						OR = {
							is_in_geographic_region = geographic_region_western_europe
							is_in_geographic_region = geographic_region_central_europe
							is_in_geographic_region = geographic_region_southern_europe
						}
					}
				}
			}

			score = {
				value = 150

				if = {
					limit = { scope:target_country = { num_allies < 1 } }
					add = 75
				}

				if = {
					limit = { scope:target_country = { is_in_war = yes } }
					add = 50
				}
			}
		}

		none = {
			trigger = {
				is_in_same_power_bloc = scope:target_country
				scope:target_country = {
					relations:root < relations_threshold:cordial
				}
			}
			score = { value = 50 }
		}
	}

	wargoal_scores = {
		join_power_bloc = {
			if = {
				limit = {
					has_secret_goal = {
						who = scope:target_country
						secret_goal = dominate
					}
					scope:target_country = {
						is_subject = no
						is_in_power_bloc = no
					}
				}
				add = 500
			}
		}

		make_protectorate = {
			if = {
				limit = {
					has_secret_goal = {
						who = scope:target_country
						secret_goal = dominate
					}
				}
				add = 100
			}
		}

		reduce_autonomy = {
			if = {
				limit = { scope:target_country = { is_subject_of = root } }
				add = 100
			}
		}

		annex_country = {
			if = {
				limit = {
					scope:target_country = { is_subject_of = root }
					scope:target_country.country_rank <= rank_value:minor_power
				}
				add = 50
			}
		}
	}

	possible = {
		is_subject = no
		c:AUS ?= this
		is_power_bloc_leader = yes
		power_bloc ?= {
			has_identity = identity:identity_holy_roman_empire
		}
	}

	weight = {
		value = 1000
	}
}
