ai_strategy_iberian_colonialism = {
	icon = "gfx/interface/icons/ai_strategy_icons/colonial_expansion.dds"
	type = diplomatic

	undesirable_infamy_level = {
		value = 25
	}

	unacceptable_infamy_level = {
		value = 50
	}

	ideological_opinion_effect_mult = {
		value = 1.0
	}

	wargoal_maneuvers_fraction = {
		value = 0

		if = {
			limit = {
				scope:enemy_country = { is_country_type = unrecognized }
			}
			add = 0.25
		}
	}

	diplomatic_play_neutrality = {
		value = 15
	}

	diplomatic_play_boldness = {
		value = -5
	}

	recklessness = {
		value = 0.0
	}

	aggression = {
		value = 0.35

		# General colonial aggression against unrecognized targets
		if = {
			limit = {
				scope:target_country = { is_country_type = unrecognized }
			}
			add = 0.75
		}

		# Broad New World interest
		if = {
			limit = {
				scope:target_country = {
					OR = {
						capital.region = sr:region_caribbean
						capital.region = sr:region_central_america
						capital.region = sr:region_brazil
						capital.region = sr:region_gran_colombia
						capital.region = sr:region_mexico
					}
				}
			}
			add = 2
		}

		# Portugal strongly favors Brazil
		if = {
			limit = {
				c:POR ?= this
				scope:target_country = {
					capital.region = sr:region_brazil
				}
			}
			add = 4
		}

		# Castile / Spain strongly favor Caribbean, Central America, Mexico
		if = {
			limit = {
				OR = {
					c:CAS ?= this
					c:SPA ?= this
				}
				scope:target_country = {
					OR = {
						capital.region = sr:region_caribbean
						capital.region = sr:region_central_america
						capital.region = sr:region_mexico
					}
				}
			}
			add = 4
		}

		# If they hate the target ideologically, let that matter a bit
		multiply = {
			value = "ai_ideological_opinion(scope:target_country)"
			multiply = -0.1
			add = 1
			max = 2.0
			min = 0.5
		}

		# Slightly restrain purely antagonistic fixation so they don't behave too erratically
		if = {
			limit = {
				has_attitude = {
					who = scope:target_country
					attitude = antagonistic
				}
			}
			multiply = 0.25
		}
	}

	building_group_weights = {
		bg_navy = 1.5
	}

	colonial_interest_ratio = {
		value = 0.35

		if = {
			limit = { navy_size >= 20 }
			add = 0.10
		}
		if = {
			limit = { navy_size >= 40 }
			add = 0.10
		}
		if = {
			limit = { navy_size >= 60 }
			add = 0.10
		}
	}

	strategic_region_scores = {
		region_caribbean = {
			if = {
				limit = {
					OR = {
						c:CAS ?= this
						c:SPA ?= this
					}
				}
				add = 300
			}
		}

		region_central_america = {
			if = {
				limit = {
					OR = {
						c:CAS ?= this
						c:SPA ?= this
					}
				}
				add = 250
			}
		}

		region_mexico = {
			if = {
				limit = {
					OR = {
						c:CAS ?= this
						c:SPA ?= this
					}
				}
				add = 225
			}
		}

		region_brazil = {
			if = {
				limit = {
					c:POR ?= this
				}
				add = 400
			}
		}

		region_gran_colombia = {
			if = {
				limit = {
					OR = {
						c:CAS ?= this
						c:SPA ?= this
						c:POR ?= this
					}
				}
				add = 150
			}
		}
	}

	secret_goal_scores = {
		dominate = {
			score = {
				if = {
					limit = {
						scope:target_country = { NOT = { is_country_type = unrecognized } }
					}
					add = -100
				}

				if = {
					limit = {
						NOT = { scope:target_country = { is_subject_of = root } }
						has_strategic_adjacency = scope:target_country
						scope:target_country = { is_country_type = unrecognized }
					}
					add = 200
				}

				if = {
					limit = {
						scope:target_country = {
							OR = {
								capital.region = sr:region_caribbean
								capital.region = sr:region_central_america
								capital.region = sr:region_brazil
								capital.region = sr:region_gran_colombia
								capital.region = sr:region_mexico
							}
						}
					}
					add = 300
				}

				if = {
					limit = {
						c:POR ?= this
						scope:target_country = {
							capital.region = sr:region_brazil
						}
					}
					add = 400
				}

				if = {
					limit = {
						OR = {
							c:CAS ?= this
							c:SPA ?= this
						}
						scope:target_country = {
							OR = {
								capital.region = sr:region_caribbean
								capital.region = sr:region_central_america
							}
						}
					}
					add = 400
				}
			}
		}

		conquer = {
			score = {
				if = {
					limit = {
						has_strategic_adjacency = scope:target_country
						scope:target_country = { is_country_type = unrecognized }
					}
					add = 100
				}

				if = {
					limit = {
						scope:target_country = { NOT = { is_country_type = unrecognized } }
					}
					add = -100
				}

				if = {
					limit = {
						scope:target_country = {
							OR = {
								capital.region = sr:region_caribbean
								capital.region = sr:region_central_america
								capital.region = sr:region_brazil
								capital.region = sr:region_gran_colombia
								capital.region = sr:region_mexico
							}
						}
					}
					add = 250
				}

				if = {
					limit = {
						c:POR ?= this
						scope:target_country = {
							capital.region = sr:region_brazil
						}
					}
					add = 350
				}

				if = {
					limit = {
						OR = {
							c:CAS ?= this
							c:SPA ?= this
						}
						scope:target_country = {
							OR = {
								capital.region = sr:region_caribbean
								capital.region = sr:region_central_america
							}
						}
					}
					add = 350
				}
			}
		}
	}

	wargoal_weights = {
		conquer_state = 2
		reduce_autonomy = 1.5
		take_treaty_port = 1.25
		make_protectorate = 1.5
		make_dominion = 1.25
	}

	possible = {
		is_subject = no
		is_country_type = recognized
		has_technology_researched = colonization

		OR = {
			c:CAS ?= this
			c:SPA ?= this
			c:POR ?= this
		}
	}

	weight = {
		value = 0

		if = {
			limit = {
				OR = {
					c:CAS ?= this
					c:SPA ?= this
					c:POR ?= this
				}
			}
			add = 200
		}

		if = {
			limit = { navy_size < 15 }
			multiply = 0.5
		}

		if = {
			limit = { navy_size >= 25 }
			multiply = 1.25
		}

		if = {
			limit = { navy_size >= 40 }
			multiply = 1.5
		}
	}
}
