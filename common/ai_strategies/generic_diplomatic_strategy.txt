ai_strategy_dynamic_opportunism = {
	icon = "gfx/interface/icons/ai_strategy_icons/maintain_power_balance.dds"

	type = diplomatic

	will_form_power_bloc = {
		OR = {
			country_rank >= rank_value:major_power
			country_rank >= rank_value:great_power
		}
	}

	undesirable_infamy_level = {
		value = 40
	}

	unacceptable_infamy_level = {
		value = 75
	}

	ideological_opinion_effect_mult = {
		value = 0.90
	}

	wargoal_maneuvers_fraction = {
		value = 0.18
	}

	diplomatic_play_neutrality = {
		value = 10
	}

	diplomatic_play_boldness = {
		value = 45
	}

	recklessness = {
		value = 0.18
	}

	diplomatic_play_support = {
		value = 0

		add = {
			value = "scope:country.modifier:country_diplomatic_reputation_add"
			desc = "DIPLOMATIC_REPUTATION_REASON"
		}

		# More willing to support allies and friends
		if = {
			limit = {
				scope:country = {
					relations:root >= relations_threshold:amicable
				}
			}
			add = {
				value = 60
				desc = "friendly_support"
			}
		}

		# Defensive instinct against aggressive nearby powers
		if = {
			limit = {
				NOT = { exists = scope:is_initiator }
				scope:enemy_country = {
					country_rank >= root.country_rank
				}
				scope:country = {
					relations:root >= relations_threshold:poor
				}
			}
			add = {
				value = 45
				desc = "contain_threat"
			}
		}

		# Less likely to support countries with bad relations
		if = {
			limit = {
				scope:country = {
					relations:root < relations_threshold:neutral
				}
			}
			add = {
				value = -60
				desc = "bad_relations"
			}
		}
	}

	aggression = {
		value = 0.65

		# Claims matter
		if = {
			limit = {
				scope:target_country = {
					any_scope_state = {
						has_claim_by = root
					}
				}
			}
			add = 0.75
		}

		# Adjacent countries are the main natural targets
		if = {
			limit = {
				can_reach_target_country = yes
				has_strategic_adjacency = scope:target_country
			}
			add = 0.60
		}

		# Small naval opportunism
		if = {
			limit = {
				can_reach_target_country = yes
			}
			add = 0.15
		}

		# Opportunism: attack countries already fighting
		if = {
			limit = {
				scope:target_country = {
					is_in_war = yes
				}
			}
			add = 0.45
		}

		# Opportunism: isolated targets
		if = {
			limit = {
				scope:target_country = {
					num_allies < 1
				}
			}
			add = 0.25
		}

		# Prefer weaker targets
		if = {
			limit = {
				scope:target_country.country_rank < root.country_rank
			}
			add = 0.30
		}

		# Prefer minor targets for subjugation / expansion
		if = {
			limit = {
				scope:target_country.country_rank <= rank_value:minor_power
			}
			add = 0.25
		}

		# Less aggression toward subjects
		if = {
			limit = {
				scope:target_country = { is_subject_of = root }
			}
			multiply = 0.20
		}

		# Reduce nonsense against distant irrelevant countries
		if = {
			limit = {
				NOT = { has_strategic_adjacency = scope:target_country }
				NOT = { can_reach_target_country = yes }
				scope:target_country = {
					NOT = {
						any_scope_state = {
							has_claim_by = root
						}
					}
					NOT = { is_subject_of = root }
				}
			}
			multiply = 0.50
		}

		multiply = {
			value = "ai_ideological_opinion(scope:target_country)"
			multiply = -0.08
			add = 1
			max = 1.75
			min = 0.60
		}

		max = 2.5
	}

	building_group_weights = {
		bg_army = 1.30
		bg_technology = 1.10
		bg_navy = 1.00
	}

	secret_goal_weights = {
		protect = 1.20
		befriend = 1.75
		reconcile = 0.60
		dominate = 1.40
		conquer = 1.35
	}

	wargoal_weights = {
		make_protectorate = 1.75
		make_dominion = 1.00
		transfer_subject = 1.00
		reduce_autonomy = 1.75
		annex_country = 1.65
		return_state = 1.50
		conquer_state = 1.85
		liberate_country = 0.90
		liberate_subject = 0.75
		force_nationalization = 0.00
	}

	secret_goal_scores = {
		protect = {
			trigger = {
				scope:target_country = {
					OR = {
						is_subject_of = root
						relations:root >= relations_threshold:amicable
					}
				}
			}
			score = {
				value = 100

				if = {
					limit = {
						scope:target_country = { is_subject_of = root }
					}
					add = 125
				}
			}
		}

		befriend = {
			trigger = {
				NOT = { scope:target_country = { is_subject_of = root } }
				scope:target_country = {
					relations:root >= relations_threshold:neutral
				}
			}
			score = {
				value = 85

				if = {
					limit = {
						scope:target_country = {
							relations:root < relations_threshold:amicable
						}
					}
					add = 85
				}

				if = {
					limit = {
						scope:target_country.country_rank >= root.country_rank
					}
					add = 35
				}
			}
		}

		reconcile = {
			trigger = {
				NOT = { scope:target_country = { is_subject_of = root } }
				scope:target_country = {
					relations:root >= relations_threshold:poor
					relations:root < relations_threshold:neutral
				}
			}
			score = {
				value = 60
			}
		}

		dominate = {
			trigger = {
				NOT = { scope:target_country = { is_subject_of = root } }
				scope:target_country.country_rank <= rank_value:minor_power
				can_reach_target_country = yes
				OR = {
					has_strategic_adjacency = scope:target_country
					scope:target_country = {
						any_scope_state = {
							has_claim_by = root
						}
					}
				}
			}
			score = {
				value = 90

				if = {
					limit = {
						has_strategic_adjacency = scope:target_country
					}
					add = 70
				}

				if = {
					limit = {
						scope:target_country = {
							is_in_war = yes
						}
					}
					add = 45
				}

				if = {
					limit = {
						scope:target_country = {
							num_allies < 1
						}
					}
					add = 25
				}
			}
		}

		conquer = {
			trigger = {
				NOT = { scope:target_country = { is_subject_of = root } }
				can_reach_target_country = yes
				scope:target_country.country_rank <= root.country_rank
				OR = {
					has_strategic_adjacency = scope:target_country
					scope:target_country = {
						any_scope_state = {
							has_claim_by = root
						}
					}
				}
			}
			score = {
				value = 100

				if = {
					limit = {
						scope:target_country = {
							any_scope_state = {
								has_claim_by = root
							}
						}
					}
					add = 120
				}

				if = {
					limit = {
						has_strategic_adjacency = scope:target_country
					}
					add = 80
				}

				if = {
					limit = {
						scope:target_country.country_rank <= rank_value:minor_power
					}
					add = 45
				}

				if = {
					limit = {
						scope:target_country = {
							is_in_war = yes
						}
					}
					add = 50
				}

				if = {
					limit = {
						scope:target_country = {
							num_allies < 1
						}
					}
					add = 25
				}
			}
		}

		none = {
			trigger = {
				scope:target_country = {
					NOT = {
						any_scope_state = {
							has_claim_by = root
						}
					}
					NOT = { is_subject_of = root }
				}
				NOT = { has_strategic_adjacency = scope:target_country }
				NOT = { can_reach_target_country = yes }
			}
			score = {
				value = -180
			}
		}
	}

	wargoal_scores = {
		make_protectorate = {
			if = {
				limit = {
					has_secret_goal = {
						who = scope:target_country
						secret_goal = dominate
					}
					scope:target_country.country_rank <= rank_value:minor_power
				}
				add = 140
			}
		}

		make_dominion = {
			if = {
				limit = {
					has_secret_goal = {
						who = scope:target_country
						secret_goal = dominate
					}
					scope:target_country.country_rank <= rank_value:minor_power
				}
				add = 80
			}
		}

		transfer_subject = {
			if = {
				limit = {
					has_secret_goal = {
						who = scope:target_country
						secret_goal = dominate
					}
				}
				add = 75
			}
		}

		reduce_autonomy = {
			if = {
				limit = {
					scope:target_country = { is_subject_of = root }
					scope:target_country = { relations:root >= relations_threshold:amicable }
				}
				add = 180
			}
		}

		annex_country = {
			if = {
				limit = {
					scope:target_country = { is_subject_of = root }
					scope:target_country.country_rank <= rank_value:minor_power
					scope:target_country = { relations:root >= relations_threshold:cordial }
				}
				add = 200
			}
		}

		return_state = {
			if = {
				limit = {
					scope:target_state = {
						has_claim_by = root
					}
				}
				add = 180
			}
		}

		conquer_state = {
			if = {
				limit = {
					has_secret_goal = {
						who = scope:target_country
						secret_goal = conquer
					}
				}
				add = 175
			}

			if = {
				limit = {
					scope:target_state = {
						has_claim_by = root
					}
				}
				add = 150
			}

			if = {
				limit = {
					has_strategic_adjacency = scope:target_country
				}
				add = 70
			}
		}

		liberate_country = {
			if = {
				limit = {
					scope:target_country.country_rank >= root.country_rank
					can_reach_target_country = yes
				}
				add = 50
			}
		}

		liberate_subject = {
			if = {
				limit = {
					scope:target_country.country_rank >= root.country_rank
					can_reach_target_country = yes
				}
				add = 40
			}
		}
	}

	possible = {
		is_subject = no
	}

	weight = {
		value = 100

		# Claims encourage assertiveness
		if = {
			limit = {
				any_country = {
					NOT = { this = root }
					any_scope_state = {
						has_claim_by = root
					}
				}
			}
			add = 60
		}

		# Adjacent opportunities matter a lot
		if = {
			limit = {
				any_country = {
					NOT = { this = root }
					NOT = { is_subject_of = root }
					has_strategic_adjacency = root
				}
			}
			add = 70
		}

		# Nearby wars create opportunities
		if = {
			limit = {
				any_country = {
					NOT = { this = root }
					can_reach_target_country = yes
					is_in_war = yes
				}
			}
			add = 35
		}

		# Nearby minors make domination logic useful
		if = {
			limit = {
				any_country = {
					NOT = { this = root }
					NOT = { is_subject_of = root }
					scope:country_rank <= rank_value:minor_power
					can_reach_target_country = yes
				}
			}
			add = 30
		}
	}
}
