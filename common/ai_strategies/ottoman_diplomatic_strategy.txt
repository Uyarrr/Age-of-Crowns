ai_strategy_ottoman_imperial_expansion = {
	icon = "gfx/interface/icons/ai_strategy_icons/territorial_expansion.dds"

	type = diplomatic

	will_form_power_bloc = {
		always = yes
	}

	undesirable_infamy_level = { value = 65 }
	unacceptable_infamy_level = { value = 95 }
	ideological_opinion_effect_mult = { value = 0.75 }
	wargoal_maneuvers_fraction = { value = 0.35 }
	diplomatic_play_neutrality = { value = 5 }
	diplomatic_play_boldness = { value = 65 }
	recklessness = { value = 1.35 }

	aggression = {
		value = 1.5

		if = {
			limit = {
				scope:target_country = { c:BYZ ?= this }
			}
			add = 4.0
		}

		if = {
			limit = {
				scope:target_country = {
					any_scope_state = {
						has_claim_by = root
					}
				}
			}
			add = 2.5
		}

		if = {
			limit = {
				NOT = { scope:target_country = { is_subject_of = root } }
				can_reach_target_country = yes
				has_strategic_adjacency = scope:target_country
			}
			add = 1.5
		}

		if = {
			limit = {
				NOT = { scope:target_country = { is_subject_of = root } }
				can_reach_target_country = yes
				scope:target_country = {
					any_scope_state = {
						OR = {
							is_in_geographic_region = geographic_region_balkans
							is_in_geographic_region = geographic_region_near_east
							is_in_geographic_region = geographic_region_arabia
							is_in_geographic_region = geographic_region_nile_basin
							is_in_geographic_region = geographic_region_north_africa
							is_in_geographic_region = geographic_region_greater_persia
						}
					}
				}
			}
			add = 1.0
		}

		if = {
			limit = {
				NOT = { scope:target_country = { is_subject_of = root } }
				can_reach_target_country = yes
				NOT = { has_strategic_adjacency = scope:target_country }
				scope:target_country = {
					NOT = {
						any_scope_state = {
							has_claim_by = root
						}
					}
				}
			}
			add = -1.5
		}

		if = {
			limit = {
				scope:target_country.country_rank < root.country_rank
			}
			add = 0.5
		}

		if = {
			limit = {
				scope:target_country = { is_subject_of = root }
			}
			multiply = 0.1
		}
	}

	building_group_weights = {
		bg_army = 1.85
		bg_navy = 1.20
	}

	secret_goal_weights = {
		conquer = 3.5
		dominate = 0.65
		protect = 0.25
		befriend = 0.0
		reconcile = 0.0
	}

	wargoal_weights = {
		conquer_state = 4.0
		return_state = 3.0
		make_protectorate = 0.65
		make_dominion = 0.25
		reduce_autonomy = 1.25
		annex_country = 1.75
		force_nationalization = 0.0
	}

	secret_goal_scores = {
		conquer = {
			trigger = {
				NOT = { scope:target_country = { is_subject_of = root } }
				can_reach_target_country = yes
				scope:target_country.country_rank <= root.country_rank
			}

			score = {
				value = 150

				if = {
					limit = {
						scope:target_country = { c:BYZ ?= this }
					}
					add = 1500
				}

				if = {
					limit = {
						scope:target_country = {
							any_scope_state = {
								has_claim_by = root
							}
						}
					}
					add = 400
				}

				if = {
					limit = {
						has_strategic_adjacency = scope:target_country
					}
					add = 175
				}

				if = {
					limit = {
						scope:target_country = {
							any_scope_state = {
								OR = {
									is_in_geographic_region = geographic_region_balkans
									is_in_geographic_region = geographic_region_near_east
									is_in_geographic_region = geographic_region_arabia
									is_in_geographic_region = geographic_region_nile_basin
									is_in_geographic_region = geographic_region_north_africa
									is_in_geographic_region = geographic_region_greater_persia
								}
							}
						}
					}
					add = 150
				}

				if = {
					limit = {
						scope:target_country.country_rank <= rank_value:minor_power
					}
					add = 100
				}

				if = {
					limit = {
						NOT = { has_strategic_adjacency = scope:target_country }
						scope:target_country = {
							NOT = {
								any_scope_state = {
									has_claim_by = root
								}
							}
						}
					}
					add = -200
				}
			}
		}

		dominate = {
			trigger = {
				NOT = { scope:target_country = { is_subject_of = root } }
				can_reach_target_country = yes
				has_strategic_adjacency = scope:target_country
				scope:target_country.country_rank <= rank_value:minor_power
			}

			score = {
				value = 40

				if = {
					limit = {
						scope:target_country = {
							NOT = {
								any_scope_state = {
									has_claim_by = root
								}
							}
						}
					}
					add = 80
				}

				if = {
					limit = {
						scope:target_country = { c:BYZ ?= this }
					}
					multiply = 0
				}
			}
		}

		protect = {
			trigger = {
				scope:target_country = { is_subject_of = root }
			}
			score = { value = 125 }
		}

		befriend = {
			trigger = {
				scope:target_country = { is_subject_of = root }
			}
			score = {
				value = 30
			}
		}

		none = {
			trigger = {
				scope:target_country = {
					any_scope_state = {
						OR = {
							is_in_geographic_region = geographic_region_balkans
							is_in_geographic_region = geographic_region_near_east
							is_in_geographic_region = geographic_region_arabia
							is_in_geographic_region = geographic_region_nile_basin
							is_in_geographic_region = geographic_region_north_africa
							is_in_geographic_region = geographic_region_greater_persia
						}
					}
				}
			}
			score = {
				value = -150
			}
		}
	}

	wargoal_scores = {
		conquer_state = {
			if = {
				limit = {
					has_secret_goal = {
						who = scope:target_country
						secret_goal = conquer
					}
				}
				add = 250
			}

			if = {
				limit = {
					scope:target_country = { c:BYZ ?= this }
				}
				add = 750
			}

			if = {
				limit = {
					scope:target_state = {
						has_claim_by = root
					}
				}
				add = 400
			}

			if = {
				limit = {
					scope:target_state = {
						any_neighbouring_state = {
							owner = root
						}
					}
				}
				add = 250
			}

			if = {
				limit = {
					scope:target_state = {
						OR = {
							is_in_geographic_region = geographic_region_balkans
							is_in_geographic_region = geographic_region_near_east
							is_in_geographic_region = geographic_region_arabia
							is_in_geographic_region = geographic_region_nile_basin
							is_in_geographic_region = geographic_region_north_africa
							is_in_geographic_region = geographic_region_greater_persia
						}
					}
				}
				add = 100
			}

			if = {
				limit = {
					NOT = {
						scope:target_state = {
							any_neighbouring_state = {
								owner = root
							}
						}
					}
					NOT = {
						scope:target_state = {
							has_claim_by = root
						}
					}
				}
				add = -175
			}
		}

		return_state = {
			if = {
				limit = {
					scope:target_state = {
						has_claim_by = root
					}
				}
				add = 350
			}

			if = {
				limit = {
					scope:target_state = {
						any_neighbouring_state = {
							owner = root
						}
					}
				}
				add = 150
			}
		}

		make_protectorate = {
			if = {
				limit = {
					has_secret_goal = {
						who = scope:target_country
						secret_goal = dominate
					}
					scope:target_country.country_rank <= rank_value:minor_power
				}
				add = 75
			}
		}

		make_dominion = {
			if = {
				limit = {
					has_secret_goal = {
						who = scope:target_country
						secret_goal = dominate
					}
					scope:target_country.country_rank <= rank_value:minor_power
				}
				add = 25
			}
		}

		reduce_autonomy = {
			if = {
				limit = {
					scope:target_country = { is_subject_of = root }
				}
				add = 125
			}
		}

		annex_country = {
			if = {
				limit = {
					scope:target_country = { is_subject_of = root }
					scope:target_country.country_rank <= rank_value:minor_power
				}
				add = 175
			}
		}
	}

	state_value = {
		value = 0

		if = {
			limit = {
				OR = {
					scope:target_state = { owner = root }
					can_reach_target_state = yes
				}
			}

			if = {
				limit = {
					scope:target_state = {
						has_claim_by = root
					}
				}
				add = 75
			}

			if = {
				limit = {
					scope:target_country = { c:BYZ ?= this }
				}
				add = 150
			}

			if = {
				limit = {
					scope:target_state = {
						any_neighbouring_state = {
							owner = root
						}
					}
				}
				add = 50
			}

			if = {
				limit = {
					scope:target_state = {
						OR = {
							is_in_geographic_region = geographic_region_balkans
							is_in_geographic_region = geographic_region_near_east
							is_in_geographic_region = geographic_region_arabia
							is_in_geographic_region = geographic_region_nile_basin
							is_in_geographic_region = geographic_region_north_africa
							is_in_geographic_region = geographic_region_greater_persia
						}
					}
				}
				add = 40
			}
		}

		min = 0
		max = 200
	}

	possible = {
		is_subject = no
		c:TUR ?= this
	}

	weight = {
		value = 100

		if = {
			limit = {
				game_date < 1475.1.1
			}
			add = 200
		}

		if = {
			limit = {
				exists = c:BYZ
				c:BYZ = {
					NOT = { is_subject_of = root }
				}
			}
			add = 300
		}

		if = {
			limit = {
				any_country = {
					NOT = { this = root }
					any_scope_state = {
						has_claim_by = root
					}
				}
			}
			add = 150
		}

		if = {
			limit = {
				any_country = {
					NOT = { this = root }
					NOT = { is_subject_of = root }
					has_strategic_adjacency = root
				}
			}
			add = 75
		}

		if = {
			limit = {
				any_country = {
					NOT = { this = root }
					NOT = { is_subject_of = root }
					has_strategic_adjacency = root
					any_scope_state = {
						OR = {
							is_in_geographic_region = geographic_region_balkans
							is_in_geographic_region = geographic_region_near_east
							is_in_geographic_region = geographic_region_arabia
							is_in_geographic_region = geographic_region_nile_basin
							is_in_geographic_region = geographic_region_north_africa
							is_in_geographic_region = geographic_region_greater_persia
						}
					}
				}
			}
			add = 75
		}
	}
}
