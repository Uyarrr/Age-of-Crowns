ai_strategy_france_royal_centralization = {
	icon = "gfx/interface/icons/ai_strategy_icons/territorial_expansion.dds"

	type = diplomatic

	will_form_power_bloc = {
		always = yes
	}

	undesirable_infamy_level = {
		value = 45
	}

	unacceptable_infamy_level = {
		value = 80
	}

	ideological_opinion_effect_mult = {
		value = 0.9
	}

	wargoal_maneuvers_fraction = {
		value = 0.25
	}

	diplomatic_play_neutrality = {
		value = 5
	}

	diplomatic_play_boldness = {
		value = 50
	}

	recklessness = {
		value = 0.3
	}

	aggression = {
		value = 1.15

		# Main rival: England
		if = {
			limit = {
				scope:target_country = { c:ENG ?= this }
			}
			add = 2.5
		}

		# Strong aggression toward adjacent targets in the French sphere
		if = {
			limit = {
				NOT = { scope:target_country = { is_subject_of = root } }
				can_reach_target_country = yes
				has_strategic_adjacency = scope:target_country
				scope:target_country = {
					OR = {
						capital.region = sr:region_france
						capital.region = sr:region_occitania
						capital.region = sr:region_rhine
					}
				}
			}
			add = 1.25
		}

		# General preference for adjacent reachable targets
		if = {
			limit = {
				NOT = { scope:target_country = { is_subject_of = root } }
				can_reach_target_country = yes
				has_strategic_adjacency = scope:target_country
			}
			add = 0.6
		}

		# Prefer weaker neighbors
		if = {
			limit = {
				scope:target_country.country_rank < root.country_rank
			}
			add = 0.4
		}

		# Tone down aggression against non-adjacent countries outside the French sphere
		if = {
			limit = {
				NOT = { scope:target_country = { is_subject_of = root } }
				can_reach_target_country = yes
				NOT = { has_strategic_adjacency = scope:target_country }
				scope:target_country = {
					NOT = {
						OR = {
							c:ENG ?= this
							capital.region = sr:region_france
							capital.region = sr:region_occitania
							capital.region = sr:region_rhine
						}
					}
				}
			}
			add = -1.0
		}

		# Much less aggression toward own subjects
		if = {
			limit = {
				scope:target_country = { is_subject_of = root }
			}
			multiply = 0.1
		}
	}

	building_group_weights = {
		bg_army = 1.75
		bg_navy = 1.15
	}

	secret_goal_weights = {
		conquer = 2.5
		dominate = 0.9
		protect = 0.5
		befriend = 0.25
		reconcile = 0.0
	}

	wargoal_weights = {
		conquer_state = 3.5
		return_state = 2.5
		make_protectorate = 1.1
		make_dominion = 0.5
		reduce_autonomy = 2.5
		annex_country = 3.0
		force_nationalization = 0.0
	}

	secret_goal_scores = {
		conquer = {
			trigger = {
				NOT = { scope:target_country = { is_subject_of = root } }
				can_reach_target_country = yes
				OR = {
					scope:target_country = { c:ENG ?= this }
					has_strategic_adjacency = scope:target_country
					scope:target_country = {
						OR = {
							capital.region = sr:region_france
							capital.region = sr:region_occitania
							capital.region = sr:region_rhine
						}
					}
				}
			}
			score = {
				value = 120

				# England is the main enemy early game
				if = {
					limit = {
						scope:target_country = { c:ENG ?= this }
					}
					add = 450
				}

				# Strong preference for French cultural sphere
				if = {
					limit = {
						scope:target_country = {
							OR = {
								capital.region = sr:region_france
								capital.region = sr:region_occitania
							}
						}
					}
					add = 200
				}

				# Rhine is still important, but slightly lower
				if = {
					limit = {
						scope:target_country = {
							capital.region = sr:region_rhine
						}
					}
					add = 125
				}

				# Adjacent targets preferred
				if = {
					limit = {
						has_strategic_adjacency = scope:target_country
					}
					add = 100
				}

				# Prefer smaller countries when expanding
				if = {
					limit = {
						scope:target_country.country_rank <= rank_value:minor_power
					}
					add = 50
				}
			}
		}

		dominate = {
			trigger = {
				NOT = { scope:target_country = { is_subject_of = root } }
				can_reach_target_country = yes
				has_strategic_adjacency = scope:target_country
				scope:target_country.country_rank <= rank_value:minor_power
				scope:target_country = {
					OR = {
						capital.region = sr:region_france
						capital.region = sr:region_occitania
						capital.region = sr:region_rhine
					}
				}
			}
			score = {
				value = 80
			}
		}

		protect = {
			trigger = {
				scope:target_country = { is_subject_of = root }
			}
			score = {
				value = 125
			}
		}

		befriend = {
			trigger = {
				scope:target_country = { is_subject_of = root }
			}
			score = {
				value = 75

				if = {
					limit = {
						scope:target_country = { relations:root < relations_threshold:amicable }
					}
					add = 75
				}
			}
		}
	}

	wargoal_scores = {
		conquer_state = {
			if = {
				limit = {
					has_secret_goal = {
						who = scope:target_country
						secret_goal = conquer
					}
				}
				add = 250
			}

			# Extra priority vs England
			if = {
				limit = {
					scope:target_country = { c:ENG ?= this }
				}
				add = 300
			}

			# Strong preference for states in the French sphere
			if = {
				limit = {
					scope:target_state = {
						OR = {
							region = sr:region_france
							region = sr:region_occitania
							region = sr:region_rhine
						}
					}
				}
				add = 125
			}

			# Prefer directly bordering states
			if = {
				limit = {
					scope:target_state = {
						any_neighbouring_state = {
							owner = root
						}
					}
				}
				add = 100
			}
		}

		return_state = {
			if = {
				limit = {
					scope:target_country = { c:ENG ?= this }
				}
				add = 300
			}

			if = {
				limit = {
					scope:target_state = {
						OR = {
							region = sr:region_france
							region = sr:region_occitania
						}
					}
				}
				add = 150
			}
		}

		make_protectorate = {
			if = {
				limit = {
					has_secret_goal = {
						who = scope:target_country
						secret_goal = dominate
					}
					scope:target_country.country_rank <= rank_value:minor_power
				}
				add = 125
			}
		}

		make_dominion = {
			if = {
				limit = {
					has_secret_goal = {
						who = scope:target_country
						secret_goal = dominate
					}
					scope:target_country.country_rank <= rank_value:minor_power
				}
				add = 50
			}
		}

		reduce_autonomy = {
			if = {
				limit = {
					scope:target_country = { is_subject_of = root }
					scope:target_country = { relations:root >= relations_threshold:amicable }
				}
				add = 250
			}
		}

		annex_country = {
			if = {
				limit = {
					scope:target_country = { is_subject_of = root }
					scope:target_country = { relations:root >= relations_threshold:cordial }
				}
				add = 300
			}
		}
	}

	possible = {
		is_subject = no
		c:FRA ?= this
	}

	weight = {
		value = 150

		# Early game: prioritize removing England
		if = {
			limit = {
				exists = c:ENG
			}
			add = 100
		}

		# If England still holds land in France or Occitania, push hard
		if = {
			limit = {
				c:ENG = {
					any_scope_state = {
						OR = {
							region = sr:region_france
							region = sr:region_occitania
						}
					}
				}
			}
			add = 150
		}

		# If subjects still exist, prioritize integration
		if = {
			limit = {
				any_country = {
					is_subject_of = root
				}
			}
			add = 75
		}
	}
}
