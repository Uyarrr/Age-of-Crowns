ai_strategy_france_royal_centralization = {
	icon = "gfx/interface/icons/ai_strategy_icons/territorial_expansion.dds"

	type = diplomatic

	will_form_power_bloc = {
		country_rank >= rank_value:major_power
	}

	undesirable_infamy_level = {
		value = 45
	}

	unacceptable_infamy_level = {
		value = 80
	}

	ideological_opinion_effect_mult = {
		value = 0.90
	}

	wargoal_maneuvers_fraction = {
		value = 0.25
	}

	diplomatic_play_neutrality = {
		value = 10
	}

	diplomatic_play_boldness = {
		value = 55
	}

	recklessness = {
		value = 0.25
	}

	aggression = {
		value = 0.85

		# Main rival: England, but not insane tunnel vision
		if = {
			limit = {
				scope:target_country = { c:ENG ?= this }
			}
			add = 0.90
		}

		# Primary sphere: France / Occitania / Rhine
		if = {
			limit = {
				scope:target_country = {
					OR = {
						capital.region = sr:region_france
						capital.region = sr:region_occitania
						capital.region = sr:region_rhine
					}
				}
			}
			add = 0.80
		}

		# Strong preference for adjacent targets
		if = {
			limit = {
				can_reach_target_country = yes
				has_strategic_adjacency = scope:target_country
			}
			add = 0.45
		}

		# Coastal / reachable targets matter too
		if = {
			limit = {
				can_reach_target_country = yes
			}
			add = 0.20
		}

		# Opportunism: strike countries already tied up in war
		if = {
			limit = {
				scope:target_country = {
					is_in_war = yes
				}
			}
			add = 0.55
		}

		# Opportunism: isolated targets
		if = {
			limit = {
				scope:target_country = {
					num_allies < 1
				}
			}
			add = 0.35
		}

		# Prefer weaker targets
		if = {
			limit = {
				scope:target_country.country_rank < root.country_rank
			}
			add = 0.35
		}

		# Reduce aggression against subjects; use integration tools instead
		if = {
			limit = {
				scope:target_country = { is_subject_of = root }
			}
			multiply = 0.20
		}

		# Reduce aggression against irrelevant distant targets, but do not cripple AI
		if = {
			limit = {
				scope:target_country = {
					NOR = {
						c:ENG ?= this
						capital.region = sr:region_france
						capital.region = sr:region_occitania
						capital.region = sr:region_rhine
						is_subject_of = root
					}
				}
				NOT = { has_strategic_adjacency = scope:target_country }
				NOT = { can_reach_target_country = yes }
			}
			multiply = 0.45
		}

		max = 3.0
	}

	building_group_weights = {
		bg_army = 1.75
		bg_technology = 1.15
	}

	secret_goal_weights = {
		conquer = 2.10
		dominate = 1.60
		protect = 0.55
		befriend = 0.40
		reconcile = 0.05
	}

	wargoal_weights = {
		conquer_state = 3.00
		return_state = 2.25
		make_protectorate = 1.75
		make_dominion = 0.90
		reduce_autonomy = 2.50
		annex_country = 2.75
		force_nationalization = 0.00
	}

	secret_goal_scores = {
		conquer = {
			trigger = {
				NOT = { scope:target_country = { is_subject_of = root } }
				can_reach_target_country = yes
				OR = {
					has_strategic_adjacency = scope:target_country
					scope:target_country = { c:ENG ?= this }
					scope:target_country = {
						OR = {
							capital.region = sr:region_france
							capital.region = sr:region_occitania
							capital.region = sr:region_rhine
						}
					}
				}
			}
			score = {
				value = 100

				# England is a major rival, but no longer absurdly over-weighted
				if = {
					limit = {
						scope:target_country = { c:ENG ?= this }
					}
					add = 220
				}

				# Strong preference for consolidation in the French sphere
				if = {
					limit = {
						scope:target_country = {
							OR = {
								capital.region = sr:region_france
								capital.region = sr:region_occitania
							}
						}
					}
					add = 180
				}

				# Rhine is a secondary expansion pressure zone
				if = {
					limit = {
						scope:target_country = {
							capital.region = sr:region_rhine
						}
					}
					add = 110
				}

				# Adjacent targets are naturally preferred
				if = {
					limit = {
						has_strategic_adjacency = scope:target_country
					}
					add = 80
				}

				# Prefer weaker targets
				if = {
					limit = {
						scope:target_country.country_rank < root.country_rank
					}
					add = 45
				}

				# Opportunistic attack windows
				if = {
					limit = {
						scope:target_country = {
							is_in_war = yes
						}
					}
					add = 65
				}

				if = {
					limit = {
						scope:target_country = {
							num_allies < 1
						}
					}
					add = 40
				}
			}
		}

		dominate = {
			trigger = {
				NOT = { scope:target_country = { is_subject_of = root } }
				can_reach_target_country = yes
				scope:target_country.country_rank <= rank_value:minor_power
				OR = {
					has_strategic_adjacency = scope:target_country
					scope:target_country = {
						OR = {
							capital.region = sr:region_france
							capital.region = sr:region_occitania
							capital.region = sr:region_rhine
						}
					}
				}
			}
			score = {
				value = 85

				if = {
					limit = {
						scope:target_country = {
							OR = {
								capital.region = sr:region_france
								capital.region = sr:region_occitania
							}
						}
					}
					add = 110
				}

				if = {
					limit = {
						scope:target_country = {
							capital.region = sr:region_rhine
						}
					}
					add = 70
				}

				if = {
					limit = {
						has_strategic_adjacency = scope:target_country
					}
					add = 50
				}

				if = {
					limit = {
						scope:target_country = {
							is_in_war = yes
						}
					}
					add = 40
				}
			}
		}

		protect = {
			trigger = {
				scope:target_country = { is_subject_of = root }
			}
			score = {
				value = 125
			}
		}

		befriend = {
			trigger = {
				scope:target_country = { is_subject_of = root }
			}
			score = {
				value = 80

				if = {
					limit = {
						scope:target_country = { relations:root < relations_threshold:amicable }
					}
					add = 70
				}
			}
		}

		none = {
			trigger = {
				scope:target_country = {
					NOR = {
						c:ENG ?= this
						capital.region = sr:region_france
						capital.region = sr:region_occitania
						capital.region = sr:region_rhine
						is_subject_of = root
					}
				}
				NOT = { has_strategic_adjacency = scope:target_country }
				NOT = { can_reach_target_country = yes }
			}
			score = {
				value = -220
			}
		}
	}

	wargoal_scores = {
		conquer_state = {
			if = {
				limit = {
					has_secret_goal = {
						who = scope:target_country
						secret_goal = conquer
					}
				}
				add = 225
			}

			# England still gets extra priority, but less extreme
			if = {
				limit = {
					scope:target_country = { c:ENG ?= this }
				}
				add = 180
			}

			# Core consolidation first
			if = {
				limit = {
					scope:target_state = {
						OR = {
							region = sr:region_france
							region = sr:region_occitania
						}
					}
				}
				add = 140
			}

			# Rhine still matters
			if = {
				limit = {
					scope:target_state = {
						region = sr:region_rhine
					}
				}
				add = 80
			}
		}

		return_state = {
			if = {
				limit = {
					scope:target_country = { c:ENG ?= this }
				}
				add = 220
			}

			if = {
				limit = {
					scope:target_state = {
						OR = {
							region = sr:region_france
							region = sr:region_occitania
						}
					}
				}
				add = 160
			}
		}

		make_protectorate = {
			if = {
				limit = {
					scope:target_country.country_rank <= rank_value:minor_power
					OR = {
						has_strategic_adjacency = scope:target_country
						scope:target_country = {
							OR = {
								capital.region = sr:region_france
								capital.region = sr:region_occitania
								capital.region = sr:region_rhine
							}
						}
					}
				}
				add = 140
			}
		}

		make_dominion = {
			if = {
				limit = {
					scope:target_country.country_rank <= rank_value:minor_power
					OR = {
						has_strategic_adjacency = scope:target_country
						scope:target_country = {
							OR = {
								capital.region = sr:region_france
								capital.region = sr:region_occitania
								capital.region = sr:region_rhine
							}
						}
					}
				}
				add = 70
			}
		}

		reduce_autonomy = {
			if = {
				limit = {
					scope:target_country = { is_subject_of = root }
					scope:target_country = { relations:root >= relations_threshold:amicable }
				}
				add = 250
			}
		}

		annex_country = {
			if = {
				limit = {
					scope:target_country = { is_subject_of = root }
					scope:target_country = { relations:root >= relations_threshold:cordial }
				}
				add = 280
			}
		}
	}

	possible = {
		is_subject = no
		c:FRA ?= this
	}

	weight = {
		value = 150

		# England remains a major strategic concern
		if = {
			limit = {
				exists = c:ENG
			}
			add = 70
		}

		# If England still holds land in the French core, push harder
		if = {
			limit = {
				c:ENG = {
					any_scope_state = {
						OR = {
							region = sr:region_france
							region = sr:region_occitania
						}
					}
				}
			}
			add = 140
		}

		# Subject integration remains important
		if = {
			limit = {
				any_country = {
					is_subject_of = root
				}
			}
			add = 80
		}

		# If opportunities exist in the French sphere, stay assertive
		if = {
			limit = {
				any_country = {
					NOT = { this = root }
					NOT = { is_subject_of = root }
					OR = {
						capital.region = sr:region_france
						capital.region = sr:region_occitania
						capital.region = sr:region_rhine
					}
				}
			}
			add = 50
		}

		# Opportunity factor: nearby war-torn region
		if = {
			limit = {
				any_country = {
					NOT = { this = root }
					can_reach_target_country = yes
					is_in_war = yes
					OR = {
						capital.region = sr:region_france
						capital.region = sr:region_occitania
						capital.region = sr:region_rhine
					}
				}
			}
			add = 35
		}
	}
}
