movement_war_of_roses_yorkist = {
	category = movement_category_ideological

	ideology = ideology_legitimist_movement
	character_ideologies = {
		ideology_royalist
		ideology_reformer
		ideology_jingoist_leader
	}

	creation_trigger = {
		OR = {
			c:ENG ?= this
			was_formed_from = ENG
		}
		has_journal_entry = je_england_2
		has_law = law_type:law_monarchy
		NOT = { has_variable = war_of_the_roses_lancaster_victory }
	}
	creation_weight = {
		value = 0
		if = {
			limit = {
				has_variable = war_of_roses_tension_current_var
				var:war_of_roses_tension_current_var >= 70
				var:war_of_roses_balance_current_var <= 50
			}
			add = 100
		}
	}
	on_created = {
		add_movement_enthusiasm_modifier = yes
	}

	disband_trigger = {
		OR = {
			NOT = { has_journal_entry = je_england_2 }
			has_variable = war_of_the_roses_lancaster_victory
		}
	}

	character_support_trigger = {
		OR = {
			has_ideology = ideology:ideology_royalist
			has_ideology = ideology:ideology_reformer
			has_ideology = ideology:ideology_jingoist_leader
			AND = {
				law_stance = {
					law = law_type:law_monarchy
					value > neutral
				}
				law_stance = {
					law = law_type:law_autocracy
					value < neutral
				}
			}
		}
	}
	character_support_weight = {
		value = 200
		if = {
			limit = { has_ideology = ideology:ideology_reformer }
			multiply = { value = 5 }
		}
	}

	pop_support_trigger = {
		NOT = { is_pop_type = slaves }
	}

	pop_support_factors = {
		movement_support_high_literacy
		movement_support_high_urbanization
		movement_support_aristocrats
		movement_support_officers
		movement_support_soldiers
		movement_support_bureaucrats
		movement_support_shopkeepers
	}

	pop_support_weight = {
		add = {
			value = 5
			desc = "POP_BASE_SUPPORT"
		}

		add = {
			desc = "URBAN_STATE"
			value = 10
			multiply = state.state_urbanization_rate
		}

		if = {
			limit = { is_pop_type = aristocrats }
			add = {
				value = 25
				desc = "POP_ARISTOCRATS"
			}
		}
		else_if = {
			limit = { is_pop_type = officers }
			add = {
				value = 18
				desc = "POP_OFFICERS"
			}
		}
		else_if = {
			limit = { is_pop_type = soldiers }
			add = {
				value = 12
				desc = "POP_SOLDIERS"
			}
		}
		else_if = {
			limit = {
				OR = {
					is_pop_type = bureaucrats
					is_pop_type = shopkeepers
				}
			}
			add = {
				value = 12
				desc = "MIDDLE_NO_ICON"
			}
		}

		if = {
			limit = {
				OR = {
					state_region = s:STATE_HOME_COUNTIES
					state_region = s:STATE_EAST_ANGLIA
					state_region = s:STATE_MIDLANDS
					state_region = s:STATE_YORKSHIRE
				}
			}
			multiply = {
				value = 1.5
				desc = "WAR_OF_THE_ROSES_YORKIST_HEARTLAND"
			}
		}

		multiply = {
			desc = "POP_LITERACY"
			value = literacy_rate
			add = 0.2
			max = 0.7
		}
	}

	revolution = {
		possible = {
			political_movement_support >= 0.05
		}

		weight = {
			value = 100
		}

		state_weight = {
			value = 1
			add = scope:clout
			if = {
				limit = {
					OR = {
						state_region = s:STATE_HOME_COUNTIES
						state_region = s:STATE_EAST_ANGLIA
						state_region = s:STATE_MIDLANDS
						state_region = s:STATE_YORKSHIRE
					}
				}
				multiply = 3
			}
			if = {
				limit = { is_capital = yes }
				multiply = 2
			}
		}
	}

	secession = {
		possible = {
			political_movement_support >= 0.05
			scope:clout <= 0
		}

		weight = {
			value = 1000
		}
	}

	additional_radicalism_factors = {
		if = {
			limit = {
				owner ?= {
					has_journal_entry = je_england_2
					var:war_of_roses_tension_current_var >= 70
				}
			}
			add = {
				value = 0.25
				desc = "WAR_OF_THE_ROSES_HIGH_TENSION"
			}
		}
		if = {
			limit = {
				owner ?= {
					has_journal_entry = je_england_2
					var:war_of_roses_balance_current_var <= 35
				}
			}
			add = {
				value = 0.20
				desc = "WAR_OF_THE_ROSES_YORK_ASCENDANT"
			}
		}
		if = {
			limit = {
				owner ?= { has_modifier = hundred_years_war_lost }
			}
			add = {
				value = 0.10
				desc = "HUNDRED_YEARS_WAR_LOST"
			}
		}
		if = {
			limit = {
				this.days_since_movement_defeated <= very_long_modifier_time
			}
			add = {
				desc = "movement_suppressed_after_civil_war"
				value = this.days_since_movement_defeated
				subtract = very_long_modifier_time
				divide = { value = very_long_modifier_time }
			}
		}
	}
}
