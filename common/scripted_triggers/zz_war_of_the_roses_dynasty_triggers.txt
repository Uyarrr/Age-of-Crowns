wotr_ruler_is_lancastrian_dynasty = {
	OR = {
		has_variable = war_of_the_roses_lancaster_victory
		AND = {
			exists = dynasty:lancaster_dynasty
			has_ruler = yes
			ruler.dynasty ?= dynasty:lancaster_dynasty
		}
		AND = {
			exists = dynasty:lancaster_dynasty
			has_ruler = no
			has_heir = yes
			heir.dynasty ?= dynasty:lancaster_dynasty
		}
	}
}
