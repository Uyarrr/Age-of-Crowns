society_age_1 = {
	era = era_1
	texture = "gfx/interface/icons/invention_icons/age_1_traditions.dds"
	category = society

	can_research = yes

	on_researched = {
		if = {
			limit = { this = c:TUR }
			add_technology_researched = ottoman_trait_1
		}
		if = {
			limit = { this = c:MOR }
			add_technology_researched = morocco_trait_1
		}
	}

	ai_weight = { value = 3 }
}
