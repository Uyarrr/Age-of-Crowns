	title = england_new_world_atlantic_coast.title
	desc = england_new_world_atlantic_coast.desc
	flavor = england_new_world_atlantic_coast.flavor

	event_image = {
		texture = "gfx/event_pictures/discovery.dds"
	}

	icon = "gfx/interface/icons/event_icons/britain_coa.dds"

	cooldown = {
		years = 999
	}

	duration = 3

	trigger = {
		has_technology_researched = colonization
		OR = {
			c:ENG ?= this
			was_formed_from = ENG
		}
		any_state = {
			count >= 4

			is_in_geographic_region = geographic_region_canada

			owner = {
				OR = {
					this = root
					is_subject_of = root
				}
			}
		}
	}

	option = {
		name = england_new_world_atlantic_coast.a
		default_option = yes

		add_involvement = {
			strategic_region = sr:region_atlantic_coast
			value = 2000
		}

		add_modifier = {
			name = atlantic_coast_expedition
			years = 10
		}
	}

	option = {
		name = england_new_world_atlantic_coast.b
	}
}
